# Live Auto-Sync Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** While the app is open, automatically read the AthleticLIVE feed and keep the Meet Day tab in lockstep — sync each finished final's results, score it (ties included), and lock it permanently.

**Architecture:** Pure logic (matching, parsing, completion detection) lives in a new dependency-free `livesync-core.js` loaded by both the browser (`<script src>`) and Node tests. The polling loop, fetch, state reconciliation, lock guards, and UI live in `index.html`. Feed is the single source of truth; synced finals lock permanently and cannot be manually edited.

**Tech Stack:** Vanilla JS (no build), Firebase RTDB REST (read-only) at `trackmeet-io.firebaseio.com/meet_73709`, Node's built-in `node:test` for unit tests, Vercel for deploy.

```js
/* Pure, dependency-free logic for live auto-sync. Loaded in the browser via
   <script src> (functions become global) and required by Node tests. */
function lsNormalizeName(x){ return (x || '').toLowerCase().replace(/[^a-z]/g, ''); }

/* Split an app entry's "Last, First" into normalized [last, first]. */
function lsAppLastFirst(a){
  if (a && a.indexOf(',') !== -1){
    const i = a.indexOf(',');
    return [lsNormalizeName(a.slice(0, i)), lsNormalizeName(a.slice(i + 1))];
  }
  return [lsNormalizeName(a), ''];
}

/* Find the index in appEntries matching a feed athlete {fn,l,tn}. Skips used[].
   Primary: last+first exact. Fallback: token-set of the two name parts. Returns -1. */
function lsMatchAthlete(feed, appEntries, used){
  const fl = lsNormalizeName(feed.l), ff = lsNormalizeName(feed.fn);
  for (let i = 0; i < appEntries.length; i++){
    if (used[i]) continue;
    const [al, af] = lsAppLastFirst(appEntries[i].a);
    if (al === fl && af === ff) return i;
  }
  const want = new Set([fl, ff].filter(Boolean));
  for (let i = 0; i < appEntries.length; i++){
    if (used[i]) continue;
    const [al, af] = lsAppLastFirst(appEntries[i].a);
    const have = new Set([al, af].filter(Boolean));
    if (have.size === want.size && [...want].every(t => have.has(t))) return i;
  }
  return -1;
}

if (typeof module !== 'undefined' && module.exports){
  module.exports = { lsNormalizeName, lsAppLastFirst, lsMatchAthlete };
}
```

```js
/* Running events: byRoundResults/<rui> is a map keyed by athlete slug.
   Identical schema for prelims and finals. Returns finishers sorted by place. */
function lsParseRunResult(node){
  if (!node || typeof node !== 'object') return [];
  const out = [];
  for (const slug of Object.keys(node)){
    const a = node[slug];
    if (!a || typeof a !== 'object' || a.p == null) continue;
    out.push({ p: Number(a.p), fn: a.fn, l: a.l, n: a.n, tn: a.tn, mark: a.m || a.er || '', q: a.q || '' });
  }
  out.sort((x, y) => x.p - y.p);
  return out;
}
```

```js
/* Combined events: combinedTable/<id>.cea[] is competitors sorted by standing.
   Each has p (place), cp (total points), a.n / a.t.n (name / team). The feed
   pre-computes points; we use the total as the display mark. */
function lsParseCombined(node){
  const cea = (node && node.cea) || [];
  return cea.map(c => {
    const ath = c.a || {};
    const full = ath.n || '';
    const sp = full.trim().split(/\s+/);

      tn: (ath.t && ath.t.n) || '',
      mark: (c.cp != null ? c.cp : 0) + ' pts',
    };
  }).filter(x => x.p > 0).sort((a, b) => a.p - b.p);
```

```js
/* Match parsed finishers to the app's entries for one event and produce the
   ordered actualResults array (each carries pl = official place, m = actual mark).
   Returns { entries, unmatched }. Unmatched finishers are reported, not dropped. */
function lsBuildActualEntries(finishers, appEntries){
  const used = new Array(appEntries.length).fill(false);
  const entries = [];
  const unmatched = [];
  for (const f of finishers){
    const i = lsMatchAthlete(f, appEntries, used);
    if (i >= 0){
      used[i] = true;
      const ent = Object.assign({}, appEntries[i]);
      ent.m = f.mark;
      if (f.p != null) ent.pl = Number(f.p);
      entries.push(ent);
    } else {
      unmatched.push({ n: ((f.fn || '') + ' ' + (f.l || '')).trim() || f.n, tn: f.tn, p: f.p });
    }
  }
  return { entries, unmatched };
}
```

```js
const LIVE_BASE = `https://trackmeet-io.firebaseio.com/meet_${LIVE_MEET_ID}`;

  async fetchJSON(path){

    if(!r.ok) throw new Error('feed '+r.status);

  /* Map a feed event_summary record to our app event key "Gender:Event".

    const gender = rec.gl;                         // "Men"/"Women"
    const name = (rec.ce || rec.n || '');
    // Find a DATA event whose gender+name resolves; tolerate feed name variants.
    const ev = DATA.events.find(e => e.gender===gender &&
      (e.name===name || name.indexOf(e.name)!==-1 || e.name.indexOf(name)!==-1));
    return ev ? evKey(ev) : null;
  },

  async poll(){
    if(!this.enabled) return;
    try{
      this.status='checking';
      const summary = await this.fetchJSON('event_summary');
      let synced = 0;
      for(const key of Object.keys(summary)){
        const rec = summary[key];
        if((rec.runm||'')!=='Finals') continue;            // finals only
        const rui = rec.rui; if(!rui) continue;
        const isCombined = (rec.ec==='Combined') || /^3[67]-/.test(rui);
        const k = this.appKeyFor(rec); if(!k) continue;
        // already locked & we keep-auto-updating: still re-check
        let finishers=[], meta={es:rec.es, lv:false};
        try{
          if(isCombined){
            const cid = (rui.split('-')[0]==='36') ? '11630' : '11629';
            const node = await this.fetchJSON(`combinedTable/${cid}`);
            meta.lv = false; meta.es = (node && node.fd===1)?'eh':rec.es;
            finishers = lsParseCombined(node);
            if(!(node && node.fd===1)) finishers = [];      // only when table final
          } else {
            const live = await this.fetchJSON(`liveRunEventList/${rui}`).catch(()=>null);
            meta.lv = !!(live && live.lv);
            const field = await this.fetchJSON(`liveFieldResults/${rui}`).catch(()=>null);
            if(field && field.st){ finishers = lsParseFieldResult(field); }
            else { const run = await this.fetchJSON(`byRoundResults/${rui}`).catch(()=>null);
                   if(run) finishers = lsParseRunResult(run); }
          }
        }catch(_){ continue; }
        if(!lsIsFinished(meta, finishers)) continue;
        const ev = DATA.events.find(e=>evKey(e)===k);
        const { entries, unmatched } = lsBuildActualEntries(finishers, ev.entries);
        if(unmatched.length) this.unmatched.push({event:k, list:unmatched});
        if(entries.length){
          synced++;
          if(this.applyWrites){
            state.actualResults[k] = entries;
            state.completedEvents[k] = true;
            state.syncLocked[k] = true;
          } else {
            console.log('[liveSync would sync]', k, entries.length+' placed', unmatched.length+' unmatched');
          }
        }
      }
      this.syncedCount = synced;
      this.status='ok'; this.lastCheck=Date.now();
      if(this.applyWrites) render();
    }catch(e){ this.status='offline'; }
  },

  start(){ this.enabled=true; this.unmatched=[]; this.poll(); this.timer=setInterval(()=>this.poll(), LIVE_POLL_MS); render(); },
  stop(){ this.enabled=false; clearInterval(this.timer); this.timer=null; render(); },
  checkNow(){ this.unmatched=[]; this.poll(); },
};
```
